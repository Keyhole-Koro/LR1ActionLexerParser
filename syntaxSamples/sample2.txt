program       : declaration_list $

declaration_list
              : declaration_list declaration
              | declaration

declaration   : type_specifier 'id' ';'
              | type_specifier 'id' '=' expression ';'

type_specifier
              : 'int'
              | 'float'
              | 'char'
              | 'double'

expression    : expression '+' term
              | expression '-' term
              | term

term          : term '*' factor
              | term '/' factor
              | factor

factor        : '(' expression ')'
              | 'id'
              | 'num'

num           : 'integer'
              | 'floating'
