program       : declarationList $

declarationList: declarationList declaration
              | declaration

declaration   : typeSpecifier 'id' ';'
              | typeSpecifier 'id' '=' expression ';'

typeSpecifier : 'int'
              | 'float'
              | 'char'
              | 'double'

expression    : expression '+' term
              | expression '-' term
              | term

term          : term '*' factor
              | term '/' factor
              | factor

factor        : '(' expression ')'
              | 'id'
              | 'num'

num           : 'integer'
              | 'floating'
